129  template <class Trainer, class Tester> double k_fold(Eigen::Ref<const Eigen::MatrixXd> X, Eigen::Ref<const Eigen::VectorXd> y, Trainer train_func, Tester test_func, const unsigned int num_folds)
130  {
131  if (X.cols() != y.size()) {
132  throw std::invalid_argument("Data size mismatch");
133  }
134  double sum_weighted_errors = 0;
135  for (unsigned int k = 0; k < num_folds; ++k) {
136  const Eigen::MatrixXd train_X(without_kth_fold_2d(X, k, num_folds));
137  const Eigen::VectorXd train_y(without_kth_fold_1d(y, k, num_folds));
138  const Eigen::MatrixXd test_X(only_kth_fold_2d(X, k, num_folds));
139  const Eigen::VectorXd test_y(only_kth_fold_1d(y, k, num_folds));
140  auto trained_model = train_func(train_X, train_y);
141  const double test_error = test_func(trained_model, test_X, test_y);
142  sum_weighted_errors += test_error * static_cast<double>(test_y.size());
143  }
144  return sum_weighted_errors / static_cast<double>(y.size());

161  template <class Trainer, class Tester> double leave_one_out(Eigen::Ref<const Eigen::MatrixXd> X, Eigen::Ref<const Eigen::VectorXd> y, Trainer train_func, Tester test_func)
162  {
163  const Eigen::Index n = y.size();
164  if (n < 2) {
165  throw std::invalid_argument("Too few data points");
166  }
167  if (X.cols() != n) {
168  throw std::invalid_argument("Data size mismatch");
169  }
170  double sum_weighted_errors = 0;
171  Eigen::MatrixXd train_X(X.rows(), n - 1);
172  Eigen::VectorXd train_y(n - 1);
173  for (Eigen::Index k = 0; k < n; ++k) {
174  if (k) {
175  train_X.leftCols(k) = X.leftCols(k);
176  train_y.head(k) = y.head(k);
177  }
178  if (k + 1 < n) {
179  const auto l = n - k - 1;
180  train_X.rightCols(l) = X.rightCols(l);
181  train_y.tail(l) = y.tail(l);
182  }
183  auto trained_model = train_func(train_X, train_y);
184  const double test_error = test_func(trained_model, X.block(0, k, X.rows(), 1), y.segment(k, 1));
185  sum_weighted_errors += test_error;
186  }
187  return sum_weighted_errors / static_cast<double>(n);
188  }
