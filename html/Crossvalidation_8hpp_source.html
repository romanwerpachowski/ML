104  template <class T> std::vector<T> without_kth_fold_1d(const std::vector<T>& data, unsigned int k, unsigned int num_folds) {
105  size_t i0, i1;
106  const size_t total_len = data.size();
107  calc_fold_indices(total_len, k, num_folds, i0, i1);
108  std::vector<T> remaining(total_len - (i1 - i0));
109  std::copy(data.begin(), data.begin() + i0, remaining.begin());
110  std::copy(data.begin() + i1, data.end(), remaining.begin() + i0);
111  return remaining;

135  {

209  template <class Trainer, class Tester> double leave_one_out(Eigen::Ref<const Eigen::MatrixXd> X, Eigen::Ref<const Eigen::VectorXd> y, Trainer train_func, Tester test_func)
210  {
211  const Eigen::Index n = y.size();
212  if (n < 2) {
213  throw std::invalid_argument("Too few data points");
214  }
215  if (X.cols() != n) {
216  throw std::invalid_argument("Data size mismatch");
217  }
218  double sum_weighted_errors = 0;
219  Eigen::MatrixXd train_X(X.rows(), n - 1);
220  Eigen::VectorXd train_y(n - 1);
221  for (Eigen::Index k = 0; k < n; ++k) {
222  if (k) {
223  train_X.leftCols(k) = X.leftCols(k);
224  train_y.head(k) = y.head(k);
225  }
226  if (k + 1 < n) {
227  const auto l = n - k - 1;
228  train_X.rightCols(l) = X.rightCols(l);
229  train_y.tail(l) = y.tail(l);
230  }
231  auto trained_model = train_func(train_X, train_y);
232  const double test_error = test_func(trained_model, X.block(0, k, X.rows(), 1), y.segment(k, 1));
233  sum_weighted_errors += test_error;
234  }
235  return sum_weighted_errors / static_cast<double>(n);
236  }
