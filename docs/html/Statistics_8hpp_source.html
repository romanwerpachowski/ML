179  {
180  if (xs.size() != ys.size()) {
181  throw std::invalid_argument("Length mismatch");
182  }
183  const auto n = static_cast<R>(xs.size());
184  if (n < 2) {
185  return std::numeric_limits<double>::quiet_NaN();
186  }
187  const R sum_x = std::accumulate(xs.begin(), xs.end(), R(0));
188  const R sum_y = std::accumulate(ys.begin(), ys.end(), R(0));
189  const auto mean_x = sum_x / n;
190  const auto mean_y = sum_y / n;
191  auto it_x = xs.begin();
192  R sum_xy(0);
193  for (auto it_y = ys.begin(); it_y != ys.end(); ++it_x, ++it_y) {
194  sum_xy += (*it_x - mean_x) * (*it_y - mean_y);
195  }
196  return sum_xy / (n - 1);
