145  bool remove_weakest_link(const double max_allowed_error_increase)
146  {
147  if (max_allowed_error_increase < 0) {
148  throw std::domain_error("Maximum allowed error increase cannot be negative");
149  }
150  if (lowest_split_nodes_.empty()) {
151  return false;
152  }
153  double lowest_error_increase = std::numeric_limits<double>::infinity();
154  SplitNode* removed = nullptr;
155  for (auto split_node_ptr : lowest_split_nodes_) {
156  assert(split_node_ptr);
157  const SplitNode& split_node = *split_node_ptr;
158  assert(split_node.lower);
159  assert(split_node.higher);
160  assert(split_node.lower->is_leaf());
161  assert(split_node.higher->is_leaf());
162  const double error_increase = split_node.error - (split_node.lower->error + split_node.higher->error);
163  if (error_increase <= lowest_error_increase) {
164  removed = split_node_ptr;
165  lowest_error_increase = error_increase;
166  }
167  }
168  assert(removed);
169  assert(lowest_error_increase >= 0);
170  if (lowest_error_increase > max_allowed_error_increase) {
171  return false;
172  }
173  SplitNode* const parent_of_removed = removed->parent;
174  auto new_leaf = std::make_unique<LeafNode>(removed->error, removed->value, parent_of_removed);
175  if (parent_of_removed) {
176  // Removing a non-root node from pruned tree.
177  bool other_is_leaf;
178  if (removed == parent_of_removed->lower.get()) {
179  parent_of_removed->lower = std::move(new_leaf);
180  other_is_leaf = parent_of_removed->higher->is_leaf();
181  } else {
182  assert(removed == parent_of_removed->higher.get());
183  parent_of_removed->higher = std::move(new_leaf);
184  other_is_leaf = parent_of_removed->lower->is_leaf();
185  }
186  // Update the set of lowest split nodes.
187  lowest_split_nodes_.erase(removed);
188  assert(!lowest_split_nodes_.count(removed));
189  if (other_is_leaf) {
190  lowest_split_nodes_.insert(parent_of_removed);
191  assert(lowest_split_nodes_.count(parent_of_removed));
192  }
193  } else {
194  // We removed the last split. Replace the pruned tree with the new leaf.
195  root_ = std::move(new_leaf);
196  lowest_split_nodes_.clear();
197  }
198  return true;
199  }
